Step1: Configure Server VLAN (PC0)
Switch>enable
Switch#configure terminal
Switch(config)#interface fastethernet0/0
Switch(config-if)#switchport access vlan 2
Switch(config-if)#exit

Step2: Configure Client VLAN (PC1)
Switch#configure terminal
Switch(config)#interface fastethernet0/1
Switch(config-if)#switchport access vlan 3
Switch(config-if)#exit

Step3: Configure Trunk Port
Switch#configure terminal
Switch(config)#interface fastethernet0/3
Switch(config-if)#switchport mode trunk

Step4: Check Router Interface
Router>enable
Router#show ip interface brief

Step5: Enable Router Interface
Router#configure terminal
Router(config)#interface fastethernet0/0
Router(config-if)#no shutdown

Step6: Create Sub-interface for VLAN 2
Router(config)#interface fastethernet0/0.1
Router(config-subif)#encapsulation dot1Q 2
Router(config-subif)#ip address 10.0.0.100 255.0.0.0
Router(config-subif)#exit

Step7: Create Sub-interface for VLAN 3
Router(config)#interface fastethernet0/0.2
Router(config-subif)#encapsulation dot1Q 3
Router(config-subif)#ip address 20.0.0.100 255.0.0.0
Router(config-subif)#exit

Step8: Configure RIP Routing
Router(config)#router rip
Router(config-router)#network 10.0.0.0
Router(config-router)#network 20.0.0.0

Step9: Configure Default Gateway
#pc0
Default Gateway
10.0.0.100

#pc1
Default Gateway
20.0.0.100

Step10: Verify Configuration on both pcs
C:\>ipconfig
#pc0
C:\>ping 20.0.0.1
#pc1
C:\>ping 10.0.0.1
