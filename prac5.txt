Step1: Configure Loopbacks and IP Addresses
#R1
config t
hostname R1
int Loopback1
ip address 192.168.1.1 255.255.255.0
exit
int Serial1/0
ip add 172.16.12.1 255.255.255.248
clock rate 128000
bandwidth 128
no shut
exit
int Serial1/1
ip add 172.16.13.1 255.255.255.248
bandwidth 64
no shut
end
write

#R2
config t
hostname R2
interface Loopback2
description R2 LAN
ip address 192.168.2.1 255.255.255.0
exit
int Serial1/0
ip add 172.16.12.2 255.255.255.248
bandwidth 128
no shut
exit
int Serial1/1
ip add 172.16.23.2 255.255.255.248
clock rate 128000
bandwidth 128
no shut
end
write

#R3

Step2: Configure EIGRP AS 1
#R1
config t
router eigrp 1
network 192.168.1.0
network 172.16.12.0 0.0.0.7
network 172.16.13.0 0.0.0.7
no auto-summary
end

#R2
config t
router eigrp 1
network 192.168.2.0
network 172.16.12.0 0.0.0.7
network 172.16.23.0 0.0.0.7
no auto-summary
end

#R3

router eigrp 1

network 172.16.13.0 0.0.0.7
network 172.16.23.0 0.0.0.7
network 172.16.34.0 0.0.0.7

#R4

Step4: Configure Policy Based Routing (PBR) on R3
#Create ACL 
config t
ip access-list standard PBR-ACL
permit 192.168.4.128 0.0.0.127
exit

#Create Route Map 
route-map R3-to-R1 permit 10
match ip address PBR-ACL
set ip next-hop 172.16.13.1
exit

#Apply Route Map 
int Serial1/2
ip policy route-map R3-to-R1
end
#Verify Route Map
show route-map

Step5: Test PBR 
#Create Debug ACL on R3

end

#Enable Debugging on R3
