Step1: Configure MPLS Core
#R1
configure terminal
hostname R1
interface Loopback0
ip address 1.1.1.1 255.255.255.255
ip ospf 1 area 0
exit
interface FastEthernet0/0
ip address 10.0.0.1 255.255.255.0
no shutdown
ip ospf 1 area 0
end

#R2
enable
configure terminal
hostname R2
interface Loopback0
ip address 2.2.2.2 255.255.255.255
ip ospf 1 area 0
exit
interface FastEthernet0/0
ip address 10.0.0.2 255.255.255.0
no shutdown
ip ospf 1 area 0
exit
interface FastEthernet1/0
ip address 10.0.1.2 255.255.255.0
no shutdown
ip ospf 1 area 0
end

#R3
enable
configure terminal
hostname R3
interface Loopback0
ip address 3.3.3.3 255.255.255.255
ip ospf 1 area 0
exit
interface FastEthernet0/0
ip address 10.0.1.3 255.255.255.0
no shutdown
ip ospf 1 area 0
end

#Configure OSPF
#R1
router ospf 1
network 1.1.1.1 0.0.0.0 area 0
network 10.0.0.0 0.0.0.255 area 0
#R2
router ospf 1
network 2.2.2.2 0.0.0.0 area 0
network 10.0.0.0 0.0.0.255 area 0
network 10.0.1.0 0.0.0.255 area 0
#R3
router ospf 1
network 3.3.3.3 0.0.0.0 area 0
network 10.0.1.0 0.0.0.255 area 0

#Verify OSPF on all routers
show ip ospf neighbor
show ip route

#Test on R1 
ping 3.3.3.3 source loopback0

Step2: Enable MPLS
#R1
configure terminal
mpls label protocol ldp
router ospf 1
mpls ldp autoconfig
end

#R2
configure terminal
mpls label protocol ldp
router ospf 1
mpls ldp autoconfig
end

#R3
configure terminal
mpls label protocol ldp
router ospf 1
mpls ldp autoconfig
end

#Verify MPLS on all routers
show mpls interfaces
show mpls ldp neighbor
show mpls forwarding-table

Step3: Configure MP-BGP Between PE Routers
#R1
router bgp 1
neighbor 3.3.3.3 remote-as 1
neighbor 3.3.3.3 update-source Loopback0
address-family vpnv4
neighbor 3.3.3.3 activate
neighbor 3.3.3.3 send-community extended
exit-address-family

#R3
router bgp 1
neighbor 1.1.1.1 remote-as 1
neighbor 1.1.1.1 update-source Loopback0
address-family vpnv4
neighbor 1.1.1.1 activate
neighbor 1.1.1.1 send-community extended
exit-address-family

#Verify MP-BGP on R1
show bgp vpnv4 unicast all summary

Step4: Configure Customer Site A
#R4
configure terminal
interface Loopback0
ip address 4.4.4.4 255.255.255.255
exit
interface FastEthernet0/0
ip address 192.168.1.4 255.255.255.0
no shutdown
exit
router ospf 2
network 4.4.4.4 0.0.0.0 area 2
network 192.168.1.0 0.0.0.255 area 2
end

#Configure VRF RED on R1
configure terminal
ip vrf RED
rd 4:4
route-target both 4:4
end

#Assign Interface to VRF on R1
configure terminal
interface FastEthernet1/0
ip vrf forwarding RED
ip address 192.168.1.1 255.255.255.0
no shutdown
end

#OSPF Inside VRF on R1
router ospf 2 vrf RED
network 192.168.1.0 0.0.0.255 area 2
#Verify on R1:
show ip route vrf RED

Step5: Configure Customer Site B
#R5
configure terminal
interface Loopback0
ip address 6.6.6.6 255.255.255.255
exit
interface FastEthernet0/0
ip address 192.168.2.6 255.255.255.0
no shutdown
exit
router ospf 2
network 6.6.6.6 0.0.0.0 area 2
network 192.168.2.0 0.0.0.255 area 2
end

#Configure VRF RED on R3
configure terminal
ip vrf RED
rd 4:4
route-target both 4:4
end
#Assign Interface on R3
configure terminal
interface FastEthernet1/0
ip vrf forwarding RED
ip address 192.168.2.1 255.255.255.0
no shutdown
end
#OSPF Inside VRF on R3
router ospf 2 vrf RED
network 192.168.2.0 0.0.0.255 area 2
#Verify on R3:
show ip route vrf RED

Step6: Exchange VRF Routes Using MP-BGP
#R1
router bgp 1
address-family ipv4 vrf RED
redistribute ospf 2
exit-address-family

#R3
router bgp 1
address-family ipv4 vrf RED
redistribute ospf 2
exit-address-family

Step7: Redistribute Back into OSPF
#R1
router ospf 2 vrf RED
redistribute bgp 1 subnets

#R3
router ospf 2 vrf RED
redistribute bgp 1 subnets

#Verification Commands
#MPLS
show mpls interfaces
show mpls ldp neighbor
show mpls forwarding-table

#VRF
show ip vrf
show ip route vrf RED

#MP-BGP
show bgp vpnv4 unicast all summary
show ip bgp vpnv4 vrf RED

#Final Testing
#From R4
ping 6.6.6.6
traceroute 6.6.6.6
