Step1: Configure R1
enable 
configure terminal
hostname R1
interface loopback0
ip add 192.168.1.1 255.255.255.0
no shutdown 
exit
int serial0/0
ip add 209.165.201.2 255.255.255.252
clock rate 128000
bandwidth 128
no shut
exit
int serial0/1
ip add 209.165.202.130 255.255.255.252
bandwidth 128
no shut
exit
end 
write memory

Step2: Configure ISP1 (R2)
enable
config t
hostname ISP1
int loopback0
ip add 209.165.200.254 255.255.255.255
exit 
int loopback1
ip add 209.165.201.30 255.255.255.255
exit
int serial0/0
ip add 209.165.201.1 255.255.255.252
bandwidth 128
no shut
exit
int serial0/1
ip add 209.165.200.225 255.255.255.252
clock rate 128000
bandwidth 128
no shut

router eigrp 1
network 209.165.200.224 0.0.0.3
network 209.165.201.0 0.0.0.31
no auto-summary
exit
ip route 192.168.1.0 255.255.255.0 209.165.201.2

end
write memory

Step4: Configure Initial Default Route on R1
config t 
ip route 0.0.0.0 0.0.0.0 209.165.201.1
end

Step5: Verify Connectivity On R1:
ping 209.165.200.254 source 192.168.1.1
ping 209.165.201.30 source 192.168.1.1
ping 209.165.202.158 source 192.168.1.1

Step6: Configure IP SLA On R1
#SLA 11 (Monitor ISP1)
config t
ip sla 11
icmp-echo 209.165.201.30
frequency 10
exit
ip sla schedule 11 life forever start-time now

#SLA 22 (Monitor ISP2)
ip sla 22
icmp-echo 209.165.202.158
frequency 10
exit
ip sla schedule 22 life forever start-time now

end

Step7: Verify IP SLA On R1:
show ip sla configuration
show ip sla statistics

Step8: Configure Tracking On R1:
config t 
track 1 rtr 11 reachability
delay down 10 up 1
exit 
track 2 rtr 22 reachability
delay down 10 up 1
exit

end

Step9: Configure Path Control Routes On R1:
config t
no ip route 0.0.0.0 0.0.0.0 209.165.201.1
#Primary Route (ISP1)
ip route 0.0.0.0 0.0.0.0 209.165.201.1 2 track 1
#Secondary Route (ISP2)
ip route 0.0.0.0 0.0.0.0 209.165.202.129 3 track 2
#Floating Backup Route
ip route 0.0.0.0 0.0.0.0 209.165.201.1 5

end
write memory

Step10: Verify Routing On R1:
show ip route
show track
show ip sla statistics

Step11: Test Failover On ISP1:
config t
interface Loopback1
shutdown
end

Step 12: Verify Automatic Failover On R1
show ip route
show ip sla statistics

Step 13: Trace Route on R1
traceroute 209.165.200.254 source 192.168.1.1

Step 14: Restore Primary Link on ISP1
config t
interface Loopback1
no shut
end

Step 15: Verify Failback on R1 
show ip route
